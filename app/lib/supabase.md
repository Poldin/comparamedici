create table public.comparator_outputs (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  input_id uuid null,
  all_info jsonb null,
  name text null,
  logo_url text null,
  n_gmapsreviews real null,
  avg_gmapsreviews_grade real null,
  last_gmapsreviews_date timestamp with time zone null,
  avg_last_quarter_median_days_gmapsreviews real null,
  website_link text null,
  facebook_link text null,
  instagram_link text null,
  tiktok_link text null,
  youtube_link text null,
  docplanner_link text null,
  n_dpreview real null,
  avg_dpreview real null,
  last_dpreview_date timestamp with time zone null,
  avg_last_quarter_days_dpreview_median real null,
  constraint comparator_outputs_pkey primary key (id),
  constraint comparator_outputs_input_id_fkey foreign KEY (input_id) references comparator_inputs (id) on update CASCADE on delete set default
) TABLESPACE pg_default;

create table public.comparator_scraper_actions (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  output jsonb null,
  action_metadata jsonb null,
  input_id uuid null,
  constraint comparator_scraper_actions_pkey primary key (id),
  constraint comparator_scraper_actions_input_id_fkey foreign KEY (input_id) references comparator_inputs (id) on update CASCADE on delete set default
) TABLESPACE pg_default;

create table public.comparator_inputs (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  link text null,
  gmaps_id text null,
  other jsonb null,
  constraint comparator_inputs_pkey primary key (id)
) TABLESPACE pg_default;
